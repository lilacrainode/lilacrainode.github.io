#include <iostream>

#include <stdio.h>

#include <cstdio>

#include <algorithm>

#include <string.h>

typedef long long LL;
LL binaryPow(LL a, LL b)
{
    if (b == 0)
    {
        return 1;
    }
    if (b &1)
    {
        return a * binaryPow(a, b - 1);
    }
    else
    {
        LL temp = binaryPow(a, b / 2);
        return temp * temp; //不要直接两个binarypow相乘，那样会增加复杂度
    }
}
int main()
{
    LL result = binaryPow(25, 6);
    printf("%lld", result);
    return 0;
}
