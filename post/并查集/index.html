const int N = 110;
int father[N];
bool isRoot[N];
void Union(int a,int b);

int findFather(int x);
int main()
{
    int n, m;
    cin >> n >> m;
    int a, b;
    init(n);
    while (m--)
    {
        scanf("%d%d", &a, &b);
        Union(a, b);
    }
    for (int i = 1; i <= n; i++)
    {
        isRoot[findFather(i)] = true;
    }
    int ans = 0;
    for (int i = 1; i <= n; i++)
    {
        if (isRoot[i])
        {
            ans++;
        }
    }
    cout << ans << endl;

}

void Union(int a, int b)
{
    int faA = findFather(a);
    int faB = findFather(b);
    if (faA != faB)
    {
        father[faA] = faB; //合并这两个并查集
        //其实这一句就是将A这个元素所在的并查集的父亲结点的父亲从其本身改为另一个并查集的父亲结点，这样就实现了两个并查集的合并
    }
    return;
}

//以下为进行路径压缩的findFather函数
int findFather(int x)
{
    int temp = x; //因为x的值会改变，所以先把x的值存起来
    while (father[x] != x)
    {
        x = father[x]; //跳出循环的时候x指向的是其所在并查集的父亲结点
    }
    while (father[temp] != temp)
    {
        int temp2 = father[temp]; //先建立一个临时变量储存temp的父亲结点
        father[temp] = x;         //再次进行遍历，并将每个元素的父亲都设置为总的父亲结点
        temp = temp2;
    }
    return x;
}

void init(int n)
{
    for (int i = 1; i <= n; i++)
    {
        father[i] = i;
        isRoot[i] = false;
    }
}
