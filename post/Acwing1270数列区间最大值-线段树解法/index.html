#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <limits.h>

using namespace std;

const int N = 100010;
int n, m, w[N];
struct Node
{
    int l, r;
    int maxv;
} tr[N * 4];

void pushup(int u)
{
    tr[u].maxv = max(tr[u << 1].maxv, tr[u << 1 | 1].maxv);
}

void build(int u, int l, int r)
{
    if (l == r)
    {
        tr[u].l = l;
        tr[u].r = r;
        tr[u].maxv = w[r];
        return;
    }
    tr[u].l = l;
    tr[u].r = r; // 这两句一定不能漏写了，否则段错误就很难debug
    // int mid = tr[u].l + tr[u].r >> 1; // mid是用来确定两边的区间的，写错了！
    int mid = l + r >> 1; // 这样写才对
    build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
    pushup(u);
}

int query(int u, int l, int r)
{
    // if(tr[u].l == tr[u].r) return tr[u].maxv; // 又写错了，和build的返回条件弄混了，应该是当前区间包含需要查找的区间才返回。l和r在整个查询过程中都没有改变
    if (tr[u].l >= l && tr[u].r <= r)
        return tr[u].maxv; // 如果当前结点表示的区间完全被需要查找的区间包含，直接返回
    int mid = tr[u].l + tr[u].r >> 1;
    int res = INT_MIN;
    if (l <= mid)
        res = query(u << 1, l, r);
    if (r > mid)
        // res = max(res, query(u << 1 | 1, mid + 1, r)); // 写错了！又弄混了，l,r在query过程中不会改变的
        res = max(res, query(u << 1 | 1, l, r));
    return res;
}

int main()
{
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
        scanf("%d", &w[i]);
    build(1, 1, n);
    int x, y;
    while (m--)
    {
        scanf("%d%d", &x, &y);
        printf("%d\n", query(1, x, y));
    }
    return 0;
}
